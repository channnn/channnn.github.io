FlappyBird

之前在网上玩过FlappyBird，想着想着便自己用JS写了一个。。

从上面的图可以得出大概的思路：

  • 配乐，底部绿色滚动条开始滚动，小鸟挥动着翅膀。

  • 点击start，把小鸟固定在一个靠左的位置，小鸟开始匀速下降。

  • 点击屏幕，小鸟匀速上升。

  • 右边开始产生长短不一的管道并向左移动。

  • 碰撞检测，通过管道时改变分数。

  • 记录每次游戏结束时的分数，与之前的分数进行比对，保存最高分数。

有了大概的思路，我们就可以进行页面布局了。

布局我就不详细说了，直接上代码吧。

然后主要来说说逻辑方面的。

  1. 首先我们要把touch.js引进项目，touchjs里包含着手机端的各种事件，这样我们就可以在手机上操作了。
    <script type="text/javascript" src="touch-0.2.14.min.js"></script>

  2. 使底部滚动条滚动，这里我是用定时器的方法。

    `

    // 无缝滚动
var oLiner=document.getElementById("line");
var lineTimer=setInterval(function(){
    if(oLiner.offsetLeft<=-343){                // 343为滚动条的宽度
        oLiner.style.left=0;
    }
    oLiner.style.left=oLiner.offsetLeft-2+"px";
},10)

`

  1. 点击开始时，我们先给小鸟定义speed、maxSpeed、type这三个属性，分别为匀速上升与下降的初始速度、最大的速度与小鸟的类型（上升与下降）。

    `

    var oBird=document.getElementById("bird");    //小鸟这个对象
var oMain=document.getElementById("main");  //页面主要区域
oBird.speed=0;                                //初始速度
oBird.maxSpeed=7;                             //小鸟下降与上升的最大速度
oBird.type="";                                //小鸟的类型

// 小鸟下降方法
function down(){
    oBird.speed+=0.5;
    oBird.type="down_bird";
    if(oBird.speed>=oBird.maxSpeed){
        oBird.speed=oBird.maxSpeed;
    }
    oBird.style.top=oBird.offsetTop+oBird.speed+"px";
    // 小鸟下降边界判断
    if(oBird.offsetTop>=oMain.clientHeight-oBird.offsetHeight){
        clearInterval(oBird.downTimer);
        clearInterval(oBird.srcTimer);
        oBird.style.top=oMain.clientHeight-oBird.offsetHeight+"px";
        gameover();
    }

}

    `

    一开始给小鸟一个上下居中于屏幕的初始位置，小鸟每次匀速下降0.5个像素，当它下降的速度等于它的最大速度时，速度就不再增加，直接把最大速度赋值给它下降的速度。当小鸟下降触碰到底部时，游戏结束。

    即当oBird.offsetTop（小鸟距离屏幕顶部的位置）的值等于或大于oMain.clientHeight（主体部分高度）-oBird.offsetHeight（小鸟自身高度）的值时，游戏结束。

  2. 点击屏幕，小鸟挥动翅膀并匀速上升。

    `

    // 小鸟上升    
var oWrap=document.getElementById("wrap");            //屏幕主体
var oAudio=document.getElementById("audio");
touch.on(oWrap,"touchstart",function(){
    clearInterval(oBird.downTimer);                    //清除下降的定时器
    clearInterval(oBird.upTimer);                    //清除上升的定时器
    oAudio.children[1].play();                        //播放上升时的音乐
    oBird.speed=oBird.maxSpeed;
    oBird.type="up_bird";                            //改变小鸟的类型
    oBird.upTimer=setInterval(function(){
        oBird.speed-=0.3;
        if(oBird.speed<=0){
            clearInterval(oBird.upTimer);
            oBird.downTimer=setInterval(down,30);    //开启下降定时器
        }
        oBird.style.top=oBird.offsetTop-oBird.speed+"px";
    },30)
})

    `

    当小鸟的上升速度等于或者小于0时，重新开启小鸟下降的定时器。

    `

    // 小鸟拍翅膀
oBird.srcTimer=setInterval(function(){
    // 改变图片下标
    oBird.index==0?oBird.index=1:oBird.index=0;
    oBird.src="img/"+oBird.type+oBird.index+".png";
},150)

    `

    用定时器不断改变小鸟图片的下标，使其能有拍打翅膀的效果。

  1. 不断产生长短不一向左移动的管道。

    `

            // pipe控制
        createPipe=setInterval(function(){
            // 生成pipe
            var upPipe=document.createElement("div");                                //创建顶部管道元素
            var downPipe=document.createElement("div");                                //创建底部管道元素
            upPipe.innerHTML='<img src="img/up_pipe.png" alt="">';                    //顶部管道元素属性
            downPipe.innerHTML='<img src="img/down_pipe.png" alt="">';                //底部管道元素属性
            upPipe.className="pipe up_pipe";
            downPipe.className="pipe down_pipe";
            oMain.appendChild(upPipe);                                                //顶部管道插入到页面
            oMain.appendChild(downPipe);                                            //底部管道插入到页面
            // 随机高度
            upPipe.style.height=parseInt(Math.random()*100)+80+"px";                //给管道一个随机的高度
            downPipe.style.height=oMain.clientHeight-upPipe.offsetHeight-140+"px";  //140为小鸟通过的区域高度
            upPipe.addScore=true;//判断是否能够加分
            // pipe移动
            upPipe.moveTimer=setInterval(function(){
                if(upPipe.offsetLeft<=-upPipe.offsetWidth){                            //当管道移出屏幕时删除结点
                    oMain.removeChild(upPipe);                                        
                    oMain.removeChild(downPipe);
                    clearInterval(upPipe.moveTimer);
                }
                upPipe.style.left=upPipe.offsetLeft-2+"px";
                downPipe.style.left=downPipe.offsetLeft-2+"px";
                // 碰撞检测
                var l1=upPipe.offsetLeft-oBird.offsetWidth;                            //小鸟与管道左边碰撞时的距离
                var l2=upPipe.offsetLeft+upPipe.offsetWidth;                        //小鸟与管道右边碰撞时的距离
                var t1=upPipe.offsetHeight;
                var t2=downPipe.offsetTop-oBird.offsetHeight;
                var t=oBird.offsetTop;
                var l=oBird.offsetLeft;
                if((l>=l1&&l<=l2&&t<=t1)||(l>=l1&&l<=l2&&t>=t2)){
                    gameover();
                }
                // 检测加分
                if(l>=l2&&upPipe.addScore){
                    oAudio.children[2].play();
                    score++;
                    upPipe.addScore=false;
                    oScore.innerHTML="";//每次都要清空
                    var sScore=String(score);
                    // alert(typeof sScore);
                    for(var i=0;i<sScore.length;i++){
                        var oImg=document.createElement("img");
                        oImg.src="img/"+sScore[i]+".jpg";
                        oScore.appendChild(oImg);
                    }
                }

            },30)
        },3000)
`

注意：有且仅当三个条件都成立时，游戏结束。因为分数是用图片的缘故，所以每次加分数时都要清空一次。

  1. 游戏结束，保存最高分到本地。

    这里使用了localstorage来储存分数，用起来非常简单，直接申明变量即可，而且没有时间限制。
